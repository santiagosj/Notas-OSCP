# Cheatsheet: Dump de base de datos

### SQLite

```bash
# Entrar a la DB
sqlite3 users.db

# Listar tablas
.tables

# Ver schema de tablas
.schema users

# Consultar todo de una tabla
SELECT * FROM users;

# Exportar toda la DB en SQL
sqlite3 users.db .dump > dump.sql

# Exportar en CSV
sqlite3 users.db "SELECT * FROM users;" > dump.csv

# Ver llaves y columnas
PRAGMA table_info(users);
```

### MySQL / MariaDB

```bash
# Conectar
mysql -u user -p -h 127.0.0.1 dbname

# Listar bases
SHOW DATABASES;

# Usar base
USE dbname;

# Tablas
SHOW TABLES;

# Describir tabla
DESCRIBE users;

# Dump con mysqldump
mysqldump -u user -p dbname > dump.sql

# Consultar directamente
mysql -u user -p -e "SELECT * FROM dbname.users;"
```

### PostgreSQL

```bash
# Conectar
psql -U user -h 127.0.0.1 dbname

# Listar bases
\l

# Usar base
\c dbname

# Listar tablas
\dt

# Ver estructura
\d users

# Dump de toda la base
pg_dump -U user -h 127.0.0.1 dbname > dump.sql
```

### MSSQL

```bash
# Conectar
sqlcmd -S 127.0.0.1 -U user -P pass

# Mostrar bases
SELECT name FROM sys.databases;

# Usar base
USE dbname;

# Mostrar tablas
SELECT * FROM information_schema.tables;

# Ver columnas
SELECT * FROM information_schema.columns WHERE table_name='users';

# Dump con bcp
bcp dbname.dbo.users out users.csv -c -t, -U user -P pass -S 127.0.0.1
```
